#include <stdio.h>

int arr1[16];
int arr2[16];

int main(){
    int n;
    scanf("%d", &n);
    
    for(int i=0; i<n; i++){
        scanf("%d", &arr1[i]);
    }
    
    for(int i=0; i<n; i++){
        scanf("%d", &arr2[i]);
    }
    
    for(int i=0; i<n; i++){
        for(int j=0; j<n; j++){
            char k;
            k=(((arr1[i]>>(n-j-1)) %2) || ((arr2[i] >> (n-j-1)) %2) ? '#' : ' ');
            printf("%c", k);
        }
        printf("\n");
    }
}
